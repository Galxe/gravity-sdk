1. 部署 node1: ./deploy_utils/deploy.sh --node node1 --mode single
2. 部署 node2: ./deploy_utils/deploy.sh --node node3 --mode single
3. 启动 node1: ./target/quick-release/gravity_cli node start --deploy-path="/tmp/node1"
4. wait 1 mins
5. 校验 validator 数量是否等于 1
6. add validator：./target/quick-release/gravity_cli validator join \
  --rpc-url http://127.0.0.1:8545 \
  --private-key 0x5c173b12be434289682782ac6f7e7bf73a6fa5a20d507e318a4bdb039b1a5f6e \
  --stake-amount 10001.0 \
  --validator-address 0x9B2C25E77a97d3e84DC0Cb7F83fb676ddC4F24b9 \
  --consensus-public-key b7a931fa544c2d1d54dee27619edfb70cc801bc599dd7a3f56f641a588cee4600b63e35d0d35fe69f2e454462b0ce9b2 \
  --validator-network-address /ip4/127.0.0.1/tcp/2026/noise-ik/99d1c7709b14777edbdbe0c602eb0186ea845ed75b01740726e581215de8625b/handshake/0 \
  --fullnode-network-address /ip4/127.0.0.1/tcp/2036/noise-ik/99d1c7709b14777edbdbe0c602eb0186ea845ed75b01740726e581215de8625b/handshake/0 \
  --aptos-address 99d1c7709b14777edbdbe0c602eb0186ea845ed75b01740726e581215de8625b
7. 启动 node3: ./target/quick-release/gravity_cli node start --deploy-path="/tmp/node3"
8. wait 2 mins
9. 校验 validator 数量是否等于 2
10. leave validator: ./target/quick-release/gravity_cli validator leave \
  --rpc-url http://127.0.0.1:8545 \
  --private-key 0x5c173b12be434289682782ac6f7e7bf73a6fa5a20d507e318a4bdb039b1a5f6e \
  --validator-address 0x9B2C25E77a97d3e84DC0Cb7F83fb676ddC4F24b9
11. wait 2 mins
12. 校验 validator 数量是否等于 1
13. 停止 node1: ./target/quick-release/gravity_cli node stop --deploy-path="/tmp/node1"
14. 停止 node1: ./target/quick-release/gravity_cli node stop --deploy-path="/tmp/node3"
